    public int getGreatestCommonDivisor(int a, int b){
        if(a == b){
            return a;
        }
        else if(a > b){
            return euclidean(a,b);
        }
        else{
            return euclidean(b,a);
        }
    }

    public int euclidean(int a, int b){
        if(b == 0){
            return a;
        }

        return euclidean(b, a % b);
    }
}
